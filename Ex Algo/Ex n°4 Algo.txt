Exercice 3 :
Nombres = [1, 2, 3]
POUR i de 0 à TAILLE de Nombres -1
    AFFICHER Valeur maximale
    AFFICHER Valeur minimale
FIN POUR

Soluce 3 :
nombres = [4, 12, 7, 19, 2]
max = nombres[0]
min = nombres[0]
FOR i de 1 à length(nombres) -1
    IF nombres[i] > max
        max = nombres[i]
    END IF
    IF nombres[i] < min
        min = nombres[i]
    END IF
END FOR
PRINT max
PRINT min
